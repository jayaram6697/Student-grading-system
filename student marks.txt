    <div class="container">
        <h2>Student Grade Calculator</h2>
        <label>Student Name:</label>
        <input type="text" id="name" placeholder="Enter name">
        
        <label>Marks (comma separated):</label>
        <input type="text" id="marks" placeholder="e.g. 85,90,70">

        <button onclick="calculate()">Calculate</button>

        <h3 id="result"></h3>
    </div>

    <script>
        function calculate() {
            let name = document.getElementById("name").value;
            let marksInput = document.getElementById("marks").value;
            let marks = marksInput.split(",").map(Number);

            let total = marks.reduce((a, b) => a + b, 0);
            let percentage = total / marks.length;
            let grade = (percentage >= 90) ? 'A' : (percentage >= 75) ? 'B' : (percentage >= 50) ? 'C' : 'F';

            document.getElementById("result").innerHTML = 
                "Name: " + name + "<br>Total: " + total + 
                "<br>Percentage: " + percentage.toFixed(2) + "%" + 
                "<br>Grade: " + grade;
        }
    </script>
